📊 Google Sheets Setup (5 min)

One-time setup. After this, one click syncs all your progress to Google Sheets.

1

Create a new Google Sheet

Go to sheets.google.com → click Blank. Name it anything you like.

2

Open Apps Script

In your Sheet: Extensions → Apps Script. A code editor opens in a new tab.

3

Paste this code (replace everything in the editor)

function doPost(e) { return handleRequest(e); } function doGet(e) { return handleRequest(e); } function handleRequest(e) { try { var raw = ''; if (e.parameter && e.parameter.data) { raw = e.parameter.data; } else if (e.postData && e.postData.contents) { raw = e.postData.contents; } var data = JSON.parse(raw); var ss = SpreadsheetApp.getActiveSpreadsheet(); var ps = ss.getSheetByName('Progress') || ss.insertSheet('Progress'); ps.clearContents(); ps.appendRow(['Week','Day','Task','Resource','Done','Confidence Stars','My Notes','Synced At']); (data.tasks||[]).forEach(function(t){ ps.appendRow([t.week,t.day,t.task,t.resource,t.done?'YES':'no',t.stars||0,t.note||'',data.syncedAt]); }); var su = ss.getSheetByName('Summary') || ss.insertSheet('Summary'); su.clearContents(); su.appendRow(['Week','Done','Total','% Complete','Synced At']); (data.summary||[]).forEach(function(s){ su.appendRow([s.week,s.done,s.total,s.pct+'%',data.syncedAt]); }); su.appendRow(['OVERALL',data.totalDone,data.totalTasks,data.overallPct+'%',data.syncedAt]); var cl = ss.getSheetByName('Checklist') || ss.insertSheet('Checklist'); cl.clearContents(); cl.appendRow(['Domain','Topic','Confident?','Synced At']); (data.checklist||[]).forEach(function(c){ cl.appendRow([c.domain,c.item,c.checked?'YES':'no',data.syncedAt]); }); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
4

Deploy as a Web App

Click Deploy → New deployment. Configure:

Type: Web app
Execute as: Me
Who has access: Anyone

Click Deploy, authorize when prompted, then copy the Web app URL (ends in /exec).

5

Paste the URL into the tracker

Close this guide, paste the URL into the green bar at the top, click Sync Now. Your Sheet gets three tabs: Progress, Summary, and Checklist. The URL saves automatically — paste it only once.

⚑ Re-deploying? If you edit the script later, always choose New deployment (not Manage/Update) to get a working URL.

📌 TMPOT's Own Study Tips (from your Topic Checklist PDF)

  • Test yourself often — after every topic, do practice questions. Don't just re-read.
  • Record yourself talking about each topic on your phone. Listen back later — it forces recall.
  • 4–6 hours/day is plenty if you're truly focused. Split it: 3 hrs morning, 3 hrs afternoon.
  • Take one full day off each week. Burnout kills retention.
  • When fatigued: move, nap, eat — don't push through. Tired studying = limited recall.
  • You don't need to know every detail. Know a little about every topic; understand concepts and interventions.
  • Always focus on: client factors → impact on OT performance → interventions.
  • Use the "Navigating Questions" PDF before any practice test to calibrate your thinking.
